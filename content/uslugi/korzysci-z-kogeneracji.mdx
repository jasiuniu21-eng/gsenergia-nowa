service_section_1_cs_section_title: "Korzyści finansowe:"
service_section_1_ilosc_kolumn: "dwie_kolumny"
service_section_1_cs_content: "<ul>  	<li>gwarancja obniżenia kosztów zużycia energii pierwotnej, co przekłada się np. na trwałe obniżenie kosztów eksploatacyjnych/kosztów produkcji,</li>  	<li> znaczne obniżenie kosztów eksploatacyjnych układów względem obecnych,</li>  	<li>obniżenie/wyeliminowanie opłat za przesył energii elektrycznej – generacja energii w miejscu poboru,</li>  	<li>wzrost bezpieczeństwa dostaw energii - zwiększenie bezpieczeństwa dostaw energii – układy równoległe</li> </ul>"
service_section_1_cs_content_2column: "<ul>  	<li>stabilne i przewidywalne koszty energii elektrycznej, cieplnej oraz chłodu,</li>  	<li>redukcja opłat za emisje CO2,</li>  	<li> zmniejszenie kosztów inwestycji w urządzenia towarzyszące, np. kotły,</li>  	<li>efektywne z punktu widzenia opodatkowania zarządzanie środkami trwałymi,</li>  	<li>zbywalne prawa majątkowe ze świadectw pochodzenia energii,</li>  	<li>możliwość zlecenia outsourcingu energetycznego.</li> </ul>"
